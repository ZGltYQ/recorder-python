   ```python
       try:
   ```

   ```python
       """Format ChromaDB query results."""
       formatted = []
       if results["ids"] and results["ids"][0]:
           for i, doc_id in enumerate(results["ids"][0]):
               formatted.append({
                   "id": doc_id,
                   "text": results["documents"][0][i] if results["documents"] else "",
                   "source": results["metadatas"][0][i].get("source", "unknown") if results["metadatas"] else "unknown",
                   "distance": results["distances"][0][i] if results["distances"] else 0.0
               })
       return formatted
   ```

   ```python
   class RAGSearch:
       """RAG search integration with LLM answering."""
       
       CITATION_FORMAT = "📄 {filename}"
       
       def __init__(self, rag_manager: RAGManager, suggestion_generator: AISuggestionGenerator):
           self.rag_manager = rag_manager
           self.suggestion_generator = suggestion_generator
       
       def format_citation(self, filename: str) -> str:
           """Format citation badge. D-03: inline badge format."""
           return f"📄 {filename}"
       
       def build_rag_prompt(self, question: str, search_results: List[Dict]) -> tuple[str, List[str]]:
           """Build prompt with retrieved context and citation sources.
           
           Returns: (prompt, list of citation strings)
           """
           if not search_results:
               return None, []
           
           context_parts = []
           citations = []
           
           for result in search_results:
               source = result.get("source", "unknown")
               text = result.get("text", "")
               if text:
                   context_parts.append(f"[Source: {source}]\n{text}")
                   citations.append(self.format_citation(source))
           
           context = "\n\n".join(context_parts)
           
           prompt = f"""You are a helpful assistant. Use the following context to answer the question.

Context:
{context}

Question: {question}

Answer based on the context above. If the context doesn't contain relevant information, say so."""
           
           return prompt, citations
       
       async def answer_with_context(self, question: str, top_k: int = 5) -> Optional[Dict]:
           """Search documents and generate AI answer with context.
           
           Returns: {answer: str, citations: List[str], has_context: bool}
           """
           # Search for relevant chunks
           search_results = await self._search_sync(question, top_k)
           
           if not search_results:
               return {"answer": None, "citations": [], "has_context": False}
           
           # Build prompt with context
           prompt, citations = self.build_rag_prompt(question, search_results)
           
           # Generate response via LLM
           # AISuggestionGenerator.generate_response expects context as List[Dict]
           context_for_llm = [
               {"text": r["text"], "source": r["source"]} 
               for r in search_results
           ]
           answer = await self.suggestion_generator.generate_response(
               question, 
               context=context_for_llm
           )
           
           return {
               "answer": answer,
               "citations": citations,
               "has_context": True
           }
       
       def _search_sync(self, query: str, top_k: int) -> List[Dict]:
           """Synchronous wrapper for search (runs in event loop)."""
           results = []
           
           def capture_results(r):
               nonlocal results
               results = r
           
           # Connect signal and trigger search
           self.rag_manager.search_complete.connect(capture_results)
           self.rag_manager.search(query, top_k)
           
           # Process Qt events to get result
           # Note: In actual async usage, this would use QEventLoop
           return results
   ```

   ```python
   def on_rag_search_clicked(self):
       """Handle manual RAG search button click."""
       # Get selected text or current transcription as question
       question = self.get_current_question_text()
       
       if not question:
           self.statusBar().showMessage("No question to search", 3000)
           return
       
       # Disable button during search
       self.btn_rag_search.setEnabled(False)
       self.statusBar().showMessage("Searching knowledge base...")
       
       # Run async search
       asyncio.create_task(self._run_rag_search(question))
   
   async def _run_rag_search(self, question: str):
       """Run RAG search asynchronously."""
       try:
           result = await self.rag_search.answer_with_context(question, top_k=3)
           
           if result["has_context"] and result["answer"]:
               # Display answer with citations
               citations = " ".join(result["citations"])
               display_text = f"{result['answer']}\n\n{citations}"
               self.display_ai_response(display_text)
           else:
               self.statusBar().showMessage("No relevant documents found", 3000)
       finally:
           self.btn_rag_search.setEnabled(True)
   
   def get_current_question_text(self) -> str:
       """Get text from current transcription or selection to use as question."""
       # Use last transcribed text if it's a question
       if hasattr(self, 'last_transcription'):
           text = self.last_transcription
           if self.ai_suggestion.is_question(text):
               return text
       return ""
   
   def display_ai_response(self, text: str):
       """Display AI response in side panel with citation formatting."""
       # Append to response area with citation formatting
       if hasattr(self, 'response_text'):
           self.response_text.append(text)
   ```
